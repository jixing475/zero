2 Add Plot

  • add box() in ui create you want show element, like plot or table
  • then use sever to calculate the element
# pkg ==============================
library(shiny)
library(shinydashboard)

# ui ==============================
ui <- dashboardPage(
  dashboardHeader(title = "My Dashboard"),
  dashboardSidebar(),
  dashboardBody(
    # add a plot
    box(plotOutput("correlation_plot"), width = 8)
  )
)

# server ==============================
server <- function(input, output){
  output$correlation_plot <- renderPlot({
    plot(iris$Sepal.Length, iris$Petal.Length)
  })
}

# run app ==============================
shinyApp(ui, server)

3 Select Input

# pkg ==============================
library(shiny)
library(shinydashboard)

# ui ==============================
ui <- dashboardPage(
  dashboardHeader(title = "My Dashboard"),
  dashboardSidebar(),
  dashboardBody(
    # add a plot
    box(plotOutput("correlation_plot"), width = 8),
    box(
      selectInput("feature", "Feature", 
                  c("Sepal.Length", "Sepal.Width", "Petal.Length", "Petal.Width", "Species")), width = 4
    )
  )
)

# server ==============================
server <- function(input, output){
  output$correlation_plot <- renderPlot({
    plot(iris$Sepal.Length, iris[[input$feature]])
  })
}

# run app ==============================
shinyApp(ui, server)

4 Menuitem And Tabitem

# pkg ==============================
library(shiny)
library(shinydashboard)

# ui ==============================
ui <- dashboardPage(
  dashboardHeader(title = "My Dashboard"),
  dashboardSidebar(
    sidebarMenu(
      menuItem("Iris", tabName = "iris", icon = icon("tree")),
      menuItem("Cars", tabName = "cars", icon = icon("car"))
    )
  ),
  dashboardBody(
    tabItems(
      tabItem("iris",
              # add a plot
              box(plotOutput("correlation_plot"), width = 8),
              box(
                selectInput("feature", "Feature", 
                            c("Sepal.Length", "Sepal.Width", "Petal.Length", "Petal.Width", "Species")), width = 4
              )),
      tabItem("cars",
              fluidPage(
                h1("cars")
              ))
    )
    

    
  )
)

# server ==============================
server <- function(input, output){
  output$correlation_plot <- renderPlot({
    plot(iris$Sepal.Length, iris[[input$feature]])
  })
}

# run app ==============================
shinyApp(ui, server)

5 DT For Table

# pkg ==============================
library(shiny)
library(shinydashboard)
library(DT)
# ui ==============================
ui <- dashboardPage(
  dashboardHeader(title = "My Dashboard"),
  dashboardSidebar(
    sidebarMenu(
      menuItem("Iris", tabName = "iris", icon = icon("tree")),
      menuItem("Cars", tabName = "cars", icon = icon("car"))
    )
  ),
  dashboardBody(
    tabItems(
      tabItem("iris",
              # add a plot
              box(plotOutput("correlation_plot"), width = 8),
              box(
                selectInput("feature", "Feature", 
                            c("Sepal.Length", "Sepal.Width", "Petal.Length", "Petal.Width", "Species")), width = 4
              )),
      tabItem("cars",
              fluidPage(
                h1("cars"),
                dataTableOutput("tableiris")
              ))
    )
    
    
    
  )
)

# server ==============================
server <- function(input, output){
  output$correlation_plot <- renderPlot({
    plot(iris$Sepal.Length, iris[[input$feature]])
  })
  
  output$tableiris <- renderDataTable(iris)
}

# run app ==============================
shinyApp(ui, server)
